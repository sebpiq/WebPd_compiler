const _tarray_ARRAYS = new Map();
// =========================== EXPORTED API
function x_tarray_createListOfArrays() {
    const tarray = [];
    return tarray;
}
function x_tarray_pushToListOfArrays(arrays, array) {
    arrays.push(array);
}
function x_tarray_getListOfArraysLength(arrays) {
    return arrays.length;
}
function x_tarray_getListOfArraysElem(arrays, index) {
    return arrays[index];
}
// =========================== TARRAY API
function tarray_create(length) {
    return new ${FloatArray}(length);
}
function tarray_get(arrayName) {
    if (!_tarray_ARRAYS.has(arrayName)) {
        throw new Error('Unknown array ' + arrayName);
    }
    return _tarray_ARRAYS.get(arrayName);
}
function tarray_has(arrayName) {
    return _tarray_ARRAYS.has(arrayName);
}
function tarray_set(arrayName, array) {
    _tarray_ARRAYS.set(arrayName, array);
    _events_ArraysChanged();
}
